package Sturt_up.com.com;

public class Operators {
    private float number1, number2;
    private char sign;
    //private String sample;
    Operators(float n1, float n2, char s) {
        number1 = n1;
        number2 = n2;
        sign = s;
    }
    Operators() {
    }

    public float result3() {
        float res = gear();
        return res;    }

    private float gear() {
        float res = 0;
        if (sign == '+' || sign == '-' || sign == '*' || sign == '/') {
            //It's ok
            if (sign == '+') res = add();
            if (sign == '-') res = subtract();
            if (sign == '*') res = multiplication();
            if (sign == '/') res = division();
        } else System.out.println("It's not ok");
        return res; }


    private float add() {
        float result = number1 + number2;
        return result; }

    private float subtract() {
        float result = number1 - number2;
        return result;}

    private float multiplication() {
        float result = number1 * number2;
        return result;}

    private float division() {
        float result = number1 / number2;
        return result;}
}
